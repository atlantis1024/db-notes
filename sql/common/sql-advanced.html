        # 表

        查看表的基本信息

        SELECT * FROM information_schema.tables
WHERE table_schema = 'test' AND table_name = 'user';

        查看表的列信息

        SELECT * FROM information_schema.columns
WHERE table_schema = 'test' AND table_name = 'user';
